import numpy as np

def run_analysis(data: dict) -> dict:
    """
    Analyzes the preprocessed sales and warehouse data to generate key findings.

    Args:
        data: A dictionary containing preprocessed pandas DataFrames.
              Expected keys include names of the DataFrames derived from
              the input file.

    Returns:
        A dictionary containing the textual summary and calculated analysis results
        as DataFrames.
    """
    analysis_results = {}

    # --- 1. Aggregate Overall Metrics ---
    
    # Assuming 'sales_data' is the main transactional table for aggregation.
    if 'sales_data' in data:
        df = data['sales_data'].copy()
        
        total_retail_sales = df['RETAIL SALES'].sum()
        total_retail_transfers = df['RETAIL TRANSFERS'].sum()
        total_warehouse_sales = df['WAREHOUSE SALES'].sum()
        
        analysis_results['overall_summary'] = {
            "Total Retail Sales": total_retail_sales,
            "Total Retail Transfers": total_retail_transfers,
            "Total Warehouse Sales": total_warehouse_sales
        }

    # --- 2. Segment Analysis by Supplier ---
    
    if 'sales_data' in data:
        df = data['sales_data'].copy()
        
        supplier_summary = df.groupby('SUPPLIER').agg(
            Total_Sales=('RETAIL SALES', 'sum'),
            Total_Transfers=('RETAIL TRANSFERS', 'sum'),
            Total_Warehouse_Sales=('WAREHOUSE SALES', 'sum'),
            Transaction_Count=('ITEM CODE', 'count')
        ).reset_index()
        
        # Calculate average metrics for supplier insight
        supplier_summary['Avg_Retail_Sales'] = supplier_summary['Total_Sales'] / supplier_summary['Transaction_Count']
        supplier_summary['Avg_Warehouse_Sales'] = supplier_summary['Total_Warehouse_Sales'] / supplier_summary['Transaction_Count']

        analysis_results['supplier_performance'] = supplier_summary
        
    # --- 3. Segment Analysis by Item Type ---
    
    if 'sales_data' in data:
        df = data['sales_data'].copy()
        
        item_type_summary = df.groupby('ITEM TYPE').agg(
            Total_Sales=('RETAIL SALES', 'sum'),
            Avg_Transfer_Rate=('RETAIL TRANSFERS', lambda x: (x.sum() / df.loc[x.index, 'RETAIL SALES']).mean() if df.loc[x.index, 'RETAIL SALES'].sum() != 0 else 0),
            Total_Warehouse_Sales=('WAREHOUSE SALES', 'sum')
        ).reset_index()

        analysis_results['item_type_performance'] = item_type_summary

    # --- 4. Identify Key Relationships (Correlation Example) ---
    
    if 'sales_data' in data and 'RETAIL SALES' in data['sales_data'].columns and 'WAREHOUSE SALES' in data['sales_data'].columns:
        df = data['sales_data'].copy()
        
        # Calculate correlation between retail sales and warehouse sales across all transactions
        correlation = df['RETAIL SALES'].corr(df['WAREHOUSE SALES'])
        analysis_results['sales_warehouse_correlation'] = {
            "Correlation (Retail vs Warehouse Sales)": correlation
        }
        
    # --- 5. Generate Textual Summary ---
    
    summary_lines = []
    
    # Overall Summary
    if 'overall_summary' in analysis_results:
        summary_lines.append("\n--- Overall Data Summary ---")
        summary_lines.append(f"Total Retail Sales across all records: {analysis_results['overall_summary']['Total Retail Sales']:,.2f}")
        summary_lines.append(f"Total Warehouse Sales across all records: {analysis_results['overall_summary']['Total Warehouse Sales']:,.2f}")
        summary_lines.append(f"The correlation between Retail Sales and Warehouse Sales is: {analysis_results['sales_warehouse_correlation']['Correlation (Retail vs Warehouse Sales)']:.4f}.")

    # Supplier Summary
    if 'supplier_performance' in analysis_results:

        # Find the top performing supplier by total sales

        summary_lines.append(f"The supplier with the highest total sales is {top_supplier['SUPPLIER']}, generating {top_supplier['Total_Sales']:,.2f} in sales.")
        summary_lines.append(f"The average retail sales per transaction for {top_supplier['SUPPLIER']} is {top_supplier['Avg_Retail_Sales']:,.2f}.")

    # Item Type Summary
