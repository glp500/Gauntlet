import pandas as pd

def preprocess(df: pd.DataFrame) -> tuple[pd.DataFrame, pd.DataFrame]:
    """
    Preprocesses the raw sales data.

    This function performs data cleaning, feature engineering, and aggregation
    to prepare the data for analysis and visualization.

    Args:
        df: The input DataFrame loaded from the CSV, expected to contain
            'YEAR', 'MONTH', 'SUPPLIER', 'ITEM CODE', 'ITEM DESCRIPTION',
            'ITEM TYPE', 'RETAIL SALES', 'RETAIL TRANSFERS', 'WAREHOUSE SALES'.

    Returns:
        A tuple containing:
        1. aggregated_sales_by_year: DataFrame aggregated by year showing total sales/transfers.
        2. aggregated_sales_by_item_type: DataFrame aggregated by item type showing total sales/transfers.
    """
    print("Starting data preprocessing...")

    # 1. Data Type Conversion and Cleaning
    # Ensure temporal data is in a proper format
    df['YEAR'] = df['YEAR'].astype(int)
    df['MONTH'] = df['MONTH'].astype(int)

    # Check for NaNs in key numerical columns and fill them with 0
    numeric_cols = ['RETAIL SALES', 'RETAIL TRANSFERS', 'WAREHOUSE SALES']
    df[numeric_cols] = df[numeric_cols].fillna(0)

    # 2. Time Series Aggregation (Focus on Year Trend)
    # Aggregate total sales and transfers by year
    sales_trends = df.groupby('YEAR').agg({
        'RETAIL SALES': 'sum',
        'RETAIL TRANSFERS': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    sales_trends.rename(columns={
        'RETAIL SALES': 'Total Retail Sales',
        'RETAIL TRANSFERS': 'Total Retail Transfers',
        'WAREHOUSE SALES': 'Total Warehouse Sales'
    }, inplace=True)

    # 3. Categorical Aggregation (Focus on Item Type)
    # Aggregate sales and transfers by Item Type
    item_type_summary = df.groupby('ITEM TYPE').agg({
        'RETAIL SALES': 'sum',
        'RETAIL TRANSFERS': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    item_type_summary.rename(columns={
        'RETAIL SALES': 'Total Sales',
        'RETAIL TRANSFERS': 'Total Transfers',
        'WAREHOUSE SALES': 'Total Warehouse Sales'
    }, inplace=True)

    print("Preprocessing complete. Aggregated data prepared.")
    return sales_trends, item_type_summary

    """
    Runs the complete analysis pipeline: loading, preprocessing, analysis, and visualization.

    Args:
        file_path: The path to the sales data file (e.g., 'Warehouse_and_Retail_Sales.csv').
    """
    print("--- Starting Data Analysis Pipeline ---")

    # 1. Data Loading
    try:
        df = data_loader.load_data(file_path)
        print(f"Data loaded successfully. Shape: {df.shape}")
    except FileNotFoundError:
        print(f"Error: File not found at {file_path}. Analysis stopped.")
        return
    except Exception as e:
        print(f"Error during data loading: {e}. Analysis stopped.")
        return

    # 2. Preprocessing
    try:
        # sales_trends: aggregated by year, item_type_summary: aggregated by item type
        sales_trends, item_type_summary = preprocessing.preprocess(df)
        print("Preprocessing successful.")
    except Exception as e:
        print(f"Error during preprocessing: {e}. Analysis stopped.")
        return

    # 3. Analysis and Summary Generation
    print("\n=========================================")
    print("         ANALYSIS SUMMARY")
    print("=========================================")

    # Textual Summary: Year-over-Year Trend Analysis
    if not sales_trends.empty:

    # Summary based on Item Type
    print("\nSales Summary by Item Type (Total Sales):")
