import pandas as pd
import numpy as np

def preprocess(df: pd.DataFrame) -> tuple[pd.DataFrame, pd.DataFrame]:
    """
    Preprocesses the raw sales data.

    This function performs data cleaning, feature engineering, and aggregation
    to prepare the data for analysis and visualization.

    Args:
        df: The input DataFrame loaded from the CSV.

    Returns:
        A tuple containing:
        1. aggregated_sales_by_year: DataFrame aggregated by year showing total sales/transfers.
        2. aggregated_sales_by_item_type: DataFrame aggregated by item type showing total sales/transfers.
    """
    print("Starting data preprocessing...")

    # 1. Data Type Conversion and Cleaning
    # Ensure temporal data is in a proper format (though often read correctly, this is a safety step)
    df['YEAR'] = df['YEAR'].astype(int)
    df['MONTH'] = df['MONTH'].astype(int)

    # Check for NaNs in key numerical columns and handle potential zero/negative values if necessary
    numeric_cols = ['RETAIL SALES', 'RETAIL TRANSFERS', 'WAREHOUSE SALES']
    df[numeric_cols] = df[numeric_cols].fillna(0) # Fill missing sales/transfer data with 0

    # 2. Time Series Aggregation (Focus on Year Trend)
    # Aggregate total sales and transfers by year
    sales_trends = df.groupby('YEAR').agg({
        'RETAIL SALES': 'sum',
        'RETAIL TRANSFERS': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    sales_trends.rename(columns={
        'RETAIL SALES': 'Total Retail Sales',
        'RETAIL TRANSFERS': 'Total Retail Transfers',
        'WAREHOUSE SALES': 'Total Warehouse Sales'
    }, inplace=True)

    # 3. Categorical Aggregation (Focus on Item Type)
    # Aggregate sales and transfers by Item Type
    item_type_summary = df.groupby('ITEM TYPE').agg({
        'RETAIL SALES': 'sum',
        'RETAIL TRANSFERS': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    item_type_summary.rename(columns={
        'RETAIL SALES': 'Total Sales',
        'RETAIL TRANSFERS': 'Total Transfers',
        'WAREHOUSE SALES': 'Total Warehouse Sales'
    }, inplace=True)

    print("Preprocessing complete. Aggregated data prepared.")
    return sales_trends, item_type_summary
