import pandas as pd
import matplotlib.pyplot as plt
from typing import List

def create_figures(data: dict, results: dict, output_dir: str) -> List[str]:
    """
    Generates an informative visualization comparing total sales across different item types
    using the pre-calculated summary results.

    Args:
        data: Dictionary containing preprocessed data. (Not strictly used for plotting here).
        results: Dictionary containing analysis results, expected to include
                 'total_sales_by_item_type'.
        output_dir: Directory path to save the figures.

    Returns:
        A list containing the absolute path(s) of the saved figure(s).
    """
    
    # Check for the required results dataframe: total sales aggregated by item type.
    if 'total_sales_by_item_type' not in results:
        return []
    
    # Use the results DataFrame provided by the analysis module
    df_item_sales = results['total_sales_by_item_type'].copy()
    
    # Identify all columns that represent sales metrics (for summing).
    # We assume columns containing 'SALES' in the upper case are quantifiable sales metrics.
    sales_cols = [col for col in df_item_sales.columns if 'SALES' in col.upper() and col != 'ITEM TYPE']
    
    if not sales_cols:
        # Cannot plot if no sales columns are found.
        return []

    # Calculate the grand total sales for each Item Type.
    # Summing across all identified sales columns (axis=1) groups the total sales contribution per item type.
    total_sales_by_type = df_item_sales.groupby('ITEM TYPE')[sales_cols].sum().reset_index()
    
    # Create the final metric for plotting: the overall total sales per item type
    total_sales_by_type['Total Sales'] = total_sales_by_type[sales_cols].sum(axis=1)

    # --- Create the figure (Bar Chart) ---
    plt.figure(figsize=(12, 7))
    
    # Use a bar chart to compare the total contribution of each item type
    plt.bar(total_sales_by_type['ITEM TYPE'], total_sales_by_type['Total Sales'], color='teal')
    
    # Add labels and title for clarity
    plt.title('Total Sales Contribution by Item Type Across All Periods', fontsize=16)
    plt.xlabel('Item Type', fontsize=12)
    plt.ylabel('Total Sales Volume (Combined Retail & Warehouse Sales)', fontsize=12)
    
    # Rotate x-axis labels for better readability
    plt.xticks(rotation=45, ha='right')
    
    # Add grid for easier reading
    plt.grid(axis='y', linestyle='--', alpha=0.6)
    
    # Ensure layout fits everything
    plt.tight_layout()
    
    # Save the figure
    figure_path = f"{output_dir}/total_sales_by_item_type_comparison.png"
    plt.savefig(figure_path)
    plt.close()
    
    return [figure_path]
