import pandas as pd
import matplotlib.pyplot as plt
from typing import list

def create_figures(data: dict, results: dict, output_dir: str) -> list[str]:
    """
    Generates an informative visualization comparing total sales across different item types
    and saves it to the specified output directory.

    Args:
        data: Dictionary containing preprocessed data (e.g., monthly summary).
        results: Dictionary containing analysis results (e.g., seasonal sales by item type).
        output_dir: Directory path to save the figures.

    Returns:
        A list containing the absolute path(s) of the saved figure(s).
    """
    
    # --- Visualization Strategy: Total Sales Comparison by Item Type ---
    
    # 1. Extract the data relevant for item type comparison
    # Assuming 'seasonal_sales_by_item_type' contains sales metrics categorized by ITEM TYPE
    if 'seasonal_sales_by_item_type' not in results:
        print("Error: 'seasonal_sales_by_item_type' not found in analysis results.")
        return []
    
    df_item_sales = results['seasonal_sales_by_item_type']
    
    # 2. Calculate the grand total sales for each Item Type across all years/months
    # We sum all numerical sales columns to get the overall contribution per item type.
    sales_cols = [col for col in df_item_sales.columns if 'SALES' in col.upper() and col not in ['ITEM TYPE']]
    
    # Group by item type and sum the sales columns
    total_sales_by_type = df_item_sales.groupby('ITEM TYPE')[sales_cols].sum().reset_index()
    
    # Calculate the overall grand total sales for each item type (for plotting purposes)
    # We will sum all available sales columns to get the maximum possible comparison metric.
    if sales_cols:
        total_sales_by_type['Total Sales'] = total_sales_by_type[sales_cols].sum(axis=1)
    else:
        print("Warning: No identifiable sales columns found for plotting.")
        return []

    # 3. Create the figure
    plt.figure(figsize=(12, 7))
    
    # Use a bar chart to compare the total contribution of each item type
    plt.bar(total_sales_by_type['ITEM TYPE'], total_sales_by_type['Total Sales'], color='skyblue')
    
    # Add labels and title for clarity
    plt.title('Total Sales Contribution by Item Type Across All Recorded Periods', fontsize=16)
    plt.xlabel('Item Type', fontsize=12)
    plt.ylabel('Total Sales Volume (Sum of Retail/Warehouse Sales)', fontsize=12)
    
    # Rotate x-axis labels if there are many item types
    plt.xticks(rotation=45, ha='right')
    
    # Add grid for easier reading
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    
    # Ensure layout fits everything
    plt.tight_layout()
    
    # 4. Save the figure
    figure_path = f"{output_dir}/sales_trend_over_time.png"
    plt.savefig(figure_path)
    plt.close()
    
    return [figure_path]
