def preprocess(data: dict) -> dict:
    """
    Cleans, aggregates, and engineers the raw sales transaction data.
    The data is summarized by Year, Month, Item Type, and Supplier,
    and a total sales column is created for downstream trend analysis.

    Args:
        data: A dictionary containing raw DataFrames. Expects 'warehouse_and_retail_sales'.

    Returns:
        A dictionary containing the aggregated DataFrame under the key 'monthly_sales_summary'.
    """
    # Ensure the input key exists
    if 'warehouse_and_retail_sales' not in data:
        raise ValueError("Input data dictionary must contain 'warehouse_and_retail_sales'.")

    df = data['warehouse_and_retail_sales'].copy()

    # Define grouping keys (dimensions) and metric columns (values)
    aggregation_keys = ['YEAR', 'MONTH', 'ITEM TYPE', 'SUPPLIER']
    metric_columns = ['RETAIL SALES', 'RETAIL TRANSFERS', 'WAREHOUSE SALES']

    # 1. Aggregate the data: Sum all sales metrics by the defined dimensions.
    # This groups transactions into monthly, item-specific summaries.
    aggregated_df = df.groupby(aggregation_keys)[metric_columns].sum().reset_index()

    # 2. Feature Engineering: Calculate the total sales volume.
    aggregated_df['TOTAL SALES'] = (
        aggregated_df['RETAIL SALES'] + 
        aggregated_df['RETAIL TRANSFERS'] + 
        aggregated_df['WAREHOUSE SALES']
    )

    # 3. Select and order the final set of columns for consistency.
    final_columns = [
        'YEAR', 
        'MONTH', 
        'TOTAL SALES', 
        'RETAIL SALES', 
        'RETAIL TRANSFERS', 
        'WAREHOUSE SALES', 
        'ITEM TYPE', 
        'SUPPLIER'
    ]
    
    # Select the necessary columns to form the final summary table
    processed_df = aggregated_df[final_columns].copy()

    # Return the data under the key specified by the shared bundle contract.
    return {
        'monthly_sales_summary': processed_df
    }
