import pandas as pd
import matplotlib.pyplot as plt
import os

def create_figures(data: dict, results: dict, output_dir: str) -> list[str]:
    """
    Generates required figures based on the provided data and analysis results.

    Args:
        data (dict): Dictionary containing preprocessed DataFrame(s) from preprocessing.py.
        results (dict): Dictionary containing analysis results from analysis.py.
        output_dir (str): The directory where figures will be saved.

    Returns:
        list[str]: A list of paths to the generated figure files.
    """
    figure_paths = []
    
    # --- Figure 1: Sales Trend Over Time ---
    
    # Check if necessary data exists for plotting
    if 'data' in data and 'RETAIL SALES' in data['data'].columns:
        df = data['data'].copy()
        
        # Aggregate sales by Year to get the trend over time
        yearly_sales = df.groupby('YEAR')['RETAIL SALES'].sum().reset_index()
        
        if not yearly_sales.empty:
            plt.figure(figsize=(12, 6))
            plt.plot(yearly_sales['YEAR'], yearly_sales['RETAIL SALES'], marker='o', linestyle='-', label='Total Retail Sales')
            plt.title('Trend of Total Retail Sales Over Years')
            plt.xlabel('Year')
            plt.ylabel('Total Retail Sales')
            plt.grid(True)
            plt.xticks(yearly_sales['YEAR'])
            plt.legend()
            
            # Define output path
            figure_name = 'sales_trend.png'
            output_path = os.path.join(output_dir, figure_name)
            plt.savefig(output_path)
            plt.close()
            figure_paths.append(output_path)

    return figure_paths
