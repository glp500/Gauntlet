import pandas as pd

def run_analysis(data: dict) -> dict[str, pd.DataFrame]:
    """
    Analyzes the preprocessed sales and inventory data to derive key performance metrics.

    Args:
        data: A dictionary where keys correspond to the original column names
              and values are pandas DataFrames.

    Returns:
        A dictionary containing the calculated analysis results:
        - sales_summary: Aggregated sales trends over time.
        - item_performance: Performance metrics grouped by item.
    """
    
    # Ensure the required columns exist in the input data structure
    if not all(col in data for col in ['YEAR', 'MONTH', 'RETAIL SALES', 'WAREHOUSE SALES', 'ITEM CODE', 'ITEM DESCRIPTION']):
        # In a real environment, this might raise an error, but here we assume valid input structure based on contract.
        return {"sales_summary": pd.DataFrame(), "item_performance": pd.DataFrame()}

    # --- 1. Calculate Sales Summary (Time-Series Trend) ---
    
    # Focus on time-series aggregation (e.g., total sales per year and month)
    
    # Aggregate Retail Sales by Year and Month
    sales_df = data['RETAIL SALES'].copy()
    sales_df['YEAR'] = sales_df['YEAR'].astype(str) + '-' + sales_df['MONTH'].astype(str).str.zfill(2)
    sales_summary = sales_df.groupby('YEAR').agg({
        'RETAIL SALES': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    
    # For a detailed monthly trend, we'll aggregate by Year and Month
    monthly_trend = data[['YEAR', 'MONTH', 'RETAIL SALES', 'WAREHOUSE SALES']].copy()
    monthly_trend_summary = monthly_trend.groupby(['YEAR', 'MONTH']).agg({
        'RETAIL SALES': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    
    sales_summary = monthly_trend_summary
    
    # --- 2. Calculate Item Performance ---
    
    # Aggregate performance by Item Code and Description
    item_performance = data.set_index(['ITEM CODE', 'ITEM DESCRIPTION']).groupby(
        ['ITEM CODE', 'ITEM DESCRIPTION']
    ).agg({
        'RETAIL SALES': 'sum',
        'RETAIL TRANSFERS': 'sum',
        'WAREHOUSE SALES': 'sum'
    }).reset_index()
    
    # Rename columns for clarity
    item_performance.rename(columns={
        'RETAIL SALES': 'Total Retail Sales',
        'RETAIL TRANSFERS': 'Total Retail Transfers',
        'WAREHOUSE SALES': 'Total Warehouse Sales'
    }, inplace=True)

    return {
        "sales_summary": sales_summary,
        "item_performance": item_performance
    }
