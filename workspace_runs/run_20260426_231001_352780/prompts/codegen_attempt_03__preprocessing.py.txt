import pandas as pd

def preprocess(data):
    """
    Preprocesses sales data for analysis.
    
    Args:
        data (pd.DataFrame): DataFrame containing sales information with columns YEAR, MONTH, SUPPLIER, ITEM_CODE, ITEM_DESCRIPTION, and ITEM_TYPE.
        
    Returns:
        dict[str, pd.DataFrame]: A dictionary with processed keys:
            - 'sales_data': Cleaned sales data with numerical conversions where necessary.
            - 'monthly_sales': Monthly aggregated sales per year.
            - 'supplier_summary': Summary statistics for each supplier (count, average sales).
    """
    # Convert YEAR and MONTH to numeric if needed
    data['YEAR'] = pd.to_numeric(data['YEAR'])
    data['MONTH'] = pd.to_numeric(data['MONTH'])

    # Ensure ITEM_CODE is integer-like or convert to categorical for analysis
    if not isinstance(data['ITEM_CODE'].dtype, int):
        data['ITEM_CODE'] = pd.Categorical(data['ITEM_CODE'], categories=range(1000000), ordered=True)
    
    # Clean up descriptions and types (e.g., strip spaces, handle nulls)
    data['ITEM_DESCRIPTION'] = data['ITEM_DESCRIPTION'].str.strip()
    data['ITEM_TYPE'] = data['ITEM_TYPE'].astype(str).map({'WINE': 'Wine', 'BEER': 'Beer', 'LIQUOR': 'Liquor'})

    # Aggregate monthly sales
    data['MONTHLY_SALES'] = (data.groupby(['YEAR', 'MONTH'])['RETAIL SALES']).sum()
    
    # Generate supplier summary statistics
    supplier_stats = (data.groupby('SUPPLIER')
                      .agg({'ITEM_TYPE': lambda x: len(x.unique()), 
                            'RETAIL TRANSFERS': 'mean',
                            'WAREHOUSE SALES': 'count'})
                      .reset_index())
                      
    return {'sales_data': data, 'monthly_sales': pd.DataFrame(data['MONTHLY_SALES'].unstack().reset_index(), columns=['YEAR', 'MONTH', 'Total Sales']), 
            'supplier_summary': supplier_stats}
